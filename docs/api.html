Sample Queries

// All cases for current user const { data: myCases } = await sb.from('cases') .select('*') .eq('user_id', (await sb.auth.getUser()).data.user.id); // Create a case const { data: newCase } = await sb.from('cases').insert({ user_id: (await sb.auth.getUser()).data.user.id, description: 'Card-not-present fraud', amount: 266527, currency: 'NGN', channel: 'Bank X' }).select().single(); // List fraud alerts (demo) const { data: alerts } = await sb.from('fraud_alerts') .select('*') .order('created_at',{ascending:false}) .limit(20);
